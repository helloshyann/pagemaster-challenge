                                            function longestWord(text) {
                                                
                                                let bookSplit = text.split(' ')
                                                let maxCharacters = 0
                                                let wordoutput = ''
                                            
                                                for (let i = 0; i < bookSplit.length; i++) {
                                                    if (bookSplit[i].length > maxCharacters) {
                                                        maxCharacters = bookSplit[i].length
                                                        wordoutput = bookSplit[i]
                                                    }
                                                }
                                            
                                                return wordoutput;
                                            }
